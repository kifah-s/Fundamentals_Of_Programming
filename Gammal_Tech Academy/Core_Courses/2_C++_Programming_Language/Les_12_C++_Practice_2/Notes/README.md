**N_1 :** "`substr` function":

```c++
/* >>> Example 3 <<< */

#include <iostream>
#include <string>

using namespace std;

int main()
{
    string name = "ahmed";

    cout << name.substr(0) << endl; // start print from index '0'.
    cout << name.substr(1) << endl; // start print from index '1'.
    cout << name.substr(2) << endl; // start print from index '2'.
    cout << name.substr(3) << endl; // start print from index '3'.
    cout << name.substr(4) << endl; // start print from index '4'.

    return 0;
}
```
